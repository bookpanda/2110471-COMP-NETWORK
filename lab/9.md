```bash
enable
config terminal
no ip domain-lookup
service password-encryption
enable secret class
banner motd #Unauthorized access is strictly prohibited.#
line con 0
password cisco
login
logging synchronous
line vty 0 4
password cisco
login

# R1
int g0/0
ip address 192.168.0.1 255.255.255.0
no shutdown
int g0/1
ip address 192.168.1.1 255.255.255.0
no shutdown
int s0/0/0
ip address 192.168.2.253 255.255.255.252
no shutdown

# R2
int s0/0/0
ip address 192.168.2.254 255.255.255.252
no shutdown
int s0/0/1
ip address 209.165.200.226 255.255.255.224
no shutdown

# ISP
int s0/0/1
ip address 209.165.200.225 255.255.255.224
no shutdown

# PC -> IP ->DHCP

# config clock rate
# R1
int s0/0/0
clock rate 128000

# R2 
int s0/0/1
clock rate 128000

# RIP
# R1
router rip
version 2
network 192.168.0.0
network 192.168.1.0
network 192.168.2.252
no auto-summary

# R2
router rip
version 2
network 192.168.2.252
default-information originate
exit
# default route to ISP on R2
ip route 0.0.0.0 0.0.0.0 209.165.200.225

# ISP (ip route dest, dest mask, next-hop)
ip route 192.168.0.0 255.255.252.0 209.165.200.226

# R2
# exclude the first 10 addresses in the DHCP pool
ip dhcp excluded-address 192.168.0.1 192.168.0.9
ip dhcp excluded-address 192.168.1.1 192.168.1.9
# create DHCP pool
ip dhcp pool R1G0
network 192.168.0.0 255.255.255.0
default-router 192.168.0.1
dns-server 209.165.200.225
exit

ip dhcp pool R1G1
network 192.168.1.0 255.255.255.0
default-router 192.168.1.1
dns-server 209.165.200.225
exit
```
